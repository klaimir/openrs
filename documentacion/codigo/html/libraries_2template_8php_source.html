621  function _build_content($region, $wrapper = NULL, $attributes = NULL)
622  {
623  $output = NULL;
624 
625  // Can't build an empty region. Exit stage left
626  if ( ! isset($region['content']) or ! count($region['content']))
627  {
628  return FALSE;
629  }
630 
631  // Possibly overwrite wrapper and attributes
632  if ($wrapper)
633  {
634  $region['wrapper'] = $wrapper;
635  }
636  if ($attributes)
637  {
638  $region['attributes'] = $attributes;
639  }
640 
641  // Open the wrapper and add attributes
642  if (isset($region['wrapper']))
643  {
644  // This just trims off the closing angle bracket. Like '<p>' to '<p'
645  $output .= substr($region['wrapper'], 0, strlen($region['wrapper']) - 1);
646 
647  // Add HTML attributes
648  if (isset($region['attributes']) && is_array($region['attributes']))
649  {
650  foreach ($region['attributes'] as $name => $value)
651  {
652  // We don't validate HTML attributes. Imagine someone using a custom XML template..
653  $output .= " $name=\"$value\"";
654  }
655  }
656 
657  $output .= ">";
658  }
659 
660  // Output the content items.
661  foreach ($region['content'] as $content)
662  {
663  $output .= $content;
664  }
665 
666  // Close the wrapper tag
667  if (isset($region['wrapper']))
668  {
669  // This just turns the wrapper into a closing tag. Like '<p>' to '</p>'
670  $output .= str_replace('<', '</', $region['wrapper']) . "\n";
671  }
672 
673  return $output;
674  }
