363  function procesar_filtros_generales($filtros)
364  {
365  // Filtro Intereses
366  if (isset($filtros['interes_id']) && $filtros['interes_id'] > 0)
367  {
368  // Intereses
369  $ids_clientes=array();
370  switch ($filtros['interes_id'])
371  {
372  case 1:
373  $this->load->model('Cliente_inmueble_model');
374  // Devuelve los ids de los clientes que tienen propiedades en venta
375  $ids_clientes = $this->Cliente_inmueble_model->get_ids_clientes(1, $filtros['tipo_interes_id']);
376  break;
377  case 2:
378  $this->load->model('Cliente_inmueble_model');
379  // Devuelve los ids de los clientes que tienen propiedades en alquiler
380  $ids_clientes = $this->Cliente_inmueble_model->get_ids_clientes(2, $filtros['tipo_interes_id']);
381  break;
382  case 3:
383  $this->load->model('Demanda_model');
384  // Devuelve los ids de los clientes que tienen demandas que buscan alquier
385  $ids_clientes = $this->Demanda_model->get_ids_clientes(2, $filtros['tipo_interes_id']);
386  break;
387  case 4:
388  $this->load->model('Demanda_model');
389  // Devuelve los ids de los clientes que tienen demandas que buscan comprar
390  $ids_clientes = $this->Demanda_model->get_ids_clientes(1, $filtros['tipo_interes_id']);
391  break;
392  default:
393  break;
394  }
395  // Comparamos los ids de cliente obtenidos
396  if(count($ids_clientes))
397  {
398  $this->db->where_in('id', $ids_clientes);
399  }
400  else
401  {
402  $this->db->where('id', 0);
403  }
404  }
405  // Filtro Pais
406  if (isset($filtros['pais_id']) && $filtros['pais_id'] >= 0)
407  {
408  $this->db->where('pais_id', $filtros['pais_id']);
409  }
410  // Filtro Provincia
411  if (isset($filtros['provincia_id']) && $filtros['provincia_id'] >= 0)
412  {
413  $this->db->where('provincia_id', $filtros['provincia_id']);
414  }
415  // Filtro Población
416  if (isset($filtros['poblacion_id']) && !empty($filtros['poblacion_id']) && $filtros['provincia_id'] >= 0)
417  {
418  $this->db->where('poblacion_id', $filtros['poblacion_id']);
419  }
420  // Filtro Agente Asignado
421  if (isset($filtros['agente_asignado_id']) && $filtros['agente_asignado_id'] >= 0)
422  {
423  if($filtros['agente_asignado_id']==0)
424  {
425  $this->db->where('agente_asignado_id IS NULL');
426  }
427  else
428  {
429  $this->db->where('agente_asignado_id', $filtros['agente_asignado_id']);
430  }
431  }
432  // Filtro estado
433  if (isset($filtros['estado_id']) && $filtros['estado_id'] >= 0)
434  {
435  $this->db->where('estado_id', $filtros['estado_id']);
436  }
437  // Filtro medio_captacion
438  if (isset($filtros['medio_captacion_id']) && $filtros['medio_captacion_id'] >= 0)
439  {
440  $this->db->where('medio_captacion_id', $filtros['medio_captacion_id']);
441  }
442  // Fechas
443  if (isset($filtros['fecha_desde']) && $filtros['fecha_desde'] != "")
444  {
445  $this->db->where('fecha_alta >=', $this->utilities->cambiafecha_form($filtros['fecha_desde']));
446  }
447  if (isset($filtros['fecha_hasta']) && $filtros['fecha_hasta'] != "")
448  {
449  $this->db->where('fecha_alta <=', $this->utilities->cambiafecha_form($filtros['fecha_hasta']));
450  }
451  }
